            # 2.选择排序

            • .思维模式：

            第一个循环是我看中的位置，第二个循环是我要找的人。

            如下是大名鼎鼎的菜鸟教程来的动图：

            最后面的黄色条状图是第一次循环，也就是我看中的位置，

            绿色是我在寻找的过程，红色是在过程中我找到的适合的人。

            • .代码：
            /*
	名称：交换函数 
	原理：异或具有的交换律和结合律的性质
	输入：数组地址，和交换数字的两个下标 
*/
void swap(int* arr,int x ,int y)
{
	int temp = 0;
	temp = arr[y];
	arr [y] = arr[x];
	arr[x] =  temp;
}

/*
	名称：输出数组
	原理：for循环遍历
	输入：数组地址 ， 数组长度 
*/
void show(int* arr,int len)
{
	printf("len = %d\r\n",len);
	
	for(int i = 0;i<len;i++ )
	{
		printf("arr[%d] = %d\r\n",i,arr[i]);
	}
 } 
/*
	名称：选择排序 
	原理：循环数组，找最小的和当前位置的数进行交换 
	输入：数组地址 ，数组长度 
*/
void selectsort(int *arr,int len)
{	
	if(arr==NULL||len<2)
	{
			return;  //边界是程序员最重要的思想。
	}
	for(int i=0;i<len-1;i++)  //第一次循环 范围是：i — N-1  
	{
		int minIndex = i;  // 把坐在位置上的人也带上寻找的过程
		for(int j = i+1;j<len;j++)
		{
			minIndex = arr[j] < arr[minIndex] ? j : minIndex;  //找到最小值。
		 }
		 swap(arr,i,minIndex); // 寻找到最小值或者说寻找这个过程结束后，
                           //才进行交换。
	}
 } 

int main(void)
{
	int arr[5] = {1,2,3,4,5}; 
	int len = sizeof(arr)/sizeof(int);
	
	swap(arr,1,4);
	
	selectsort(arr,len);
	show(arr,len);
	
	return 0;
 }

            • # 稳定性

              不稳的原因是，比如我需要排列的数组是5，8，5，2，9。

              标记两个5为5(1)，5(2)，

              数组就会变成5(1)，8，5(2)，2，9。

              根据我上面的代码逻辑，第一个外循环结束后就会变成

              2，8，5(2)，5(1)，4。

              相同的数，和原来的顺序不一样了，这就叫做稳定性。稳定性的意义跳转

            • # 描述

            不常用的原因是有比他更好的存在，

            时间复杂度来说，稳定的O(N2N^{2}N2)是最慢的算法，

            常数操作的时间也没有插入来的快。

            倒是有一个应用场景非常符合它，

            就是在一个数组中选出前几个的场景。

            • # 需要注意的问题

            截取最核心的代码过来：

            for(int i=0;i<len-1;i++)  //第一次循环 范围是：i — N-1  
	{
		int minIndex = i;  // 把坐在位置上的人也带上寻找的过程
		for(int j = i+1;j<len;j++)
		{
			minIndex = arr[j] < arr[minIndex] ? j : minIndex;  //找到最小值。
		 }
		 swap(arr,i,minIndex); // 寻找到最小值或者说寻找这个过程结束后，才进行交换。
	}

            第一个循环中我把范围定义为：i ~ N-1，

            如果我定义为：i ~ N，在第二个循环的最后面j就会变成N+1。

            内存警告。你访问了arr[N+1]的内容，但是你却只定义了arr[N+1]。

            如果你觉得这句话很奇怪，说明你不知道定义arr[N+1]只能够访问arr[N]的内容。敲重点
