                                        # 1.各种排序的结论

                                        名称时间常数空间常数稳定性描述需要注意的问题具体时间数字
                                        归并排序O(N2N^{2}N2)O(N)稳定需要稳定情况下最好的算法merge的理解O(NlogN)

                                        # 2.归并排序

                                        • .思维模式：

                                        无脑双分，然后归并。

                                        • .代码：
                                        void merge(int* arr,int L,int mid,int R);

void dichotomy(int* arr,int L,int R)
{	
	if(L==R)
	{
		return;
	}
	int mid = L+ ((R-L)>>1);

	dichotomy(arr,L,mid);
	dichotomy(arr,mid+1,R);
	merge(arr,L,mid,R);
	return; 
}

void merge(int* arr,int L,int mid,int R)
{
	int help[R-L+1];
	int i = 0;
	int help_len = sizeof(help)/sizeof(help[0]);
	
	int p1 = L;
	int p2 = mid+1;
	while(p1 <=mid && p2<=R)
	{
		help[i++] = arr[p1] > arr[p2] ? arr[p1++] : arr[p2++];
	}
	while(p1<=mid)
	{
		help[i++] = arr[p1++];
	}
	while(p2<=R)
	{
		help[i++] = arr[p2++];
	}
	
	for(int i=0;i<help_len;i++)
	{
		arr[L+i] = help[i];
	}
}


int main(void)
{
	int arr[7] = {1,2,3,4,9,6,8}; 
	int len = sizeof(arr)/sizeof(int);
	
	dichotomy(arr,0,len-1); 

	show(arr,len);

	return 0;
 }

                                        # 稳定性

                                        稳定，因为分开，需要交换左边等于右边是不会交换的。所以相同的时候不会交换，相同的数永远都在左边。

                                        # 描述

                                          1. 具有稳定性最快的算法，代价是需要格外的空间开销。
                                          1. 因为用到了数组本身的顺序，所以时间不是O(N2)O(N^{2})O(N2)

                                        # 需要注意的问题

                                          1. 在最后的merge中，是需要把两个数组整合起来的，所以mid也需要传进去，用作区分两个数组的标记。
                                          1. 递归的接受可能再数组的任意位置，所以边界要设好。

                                        # 2.归并排序-链表

                                        • .思维模式：

                                        和数组的是一致的，但是链表的细节和归并的理解需要更加准确。

                                        • .代码：快慢指针找链表中点的办法：
                                        Node* get_mid(list head)
{
	list Fast = head;
	list Low = head;
	
	while(Fast->next)
	{
		if(Fast->next->next)
		{
			Fast = Fast->next->next;
			Low = Low->next;
		}
		else if(Fast->next->next==NULL)
		{
			Fast = Fast->next;
		}
	}
	return Low;
 }

                                        二分递归函数：

                                        list mergeSort_list(list head)
{
	//the boundary of the list.
	if(head->next==NULL||head==NULL) return head;
	
	//find the middle of the list
	list mid = get_mid(head);
	list R = mid->next;
	mid->next = NULL;
	//get the list divided in half
	list L1 = mergeSort_list(head); list L2 = mergeSort_list(R);
	
	return merge(L1,L2);;
 }

                                        # 3.快速排序

                                        • .思维模式：

                                          快速排序open in new window

                                        核心思想就是只找到指定基准值的位置，在数组中表现为交换。

                                        数组可以两边往中间走。

                                        • .代码：
                                        void QuickSort(int *arr,int L,int R)
{	
	if(arr==NULL)	return;

	if(L>=R) 	return ;

	int i = L; //这里绝对不可以+1
	int j = R;
	int temp = arr[L]; //set the arr[L] as the srot number
	
	while(i != j)
	{	
		while(arr[j]>=temp && j>i) j--;
		while(arr[i]<=temp && j>i) i++;
		if(i<j) swap(arr,i,j);
	}
	arr[L] = arr[i];
	arr[i] = temp;
	
	QuickSort(arr,i+1,R); 
	QuickSort(arr,L,i-1);
 }

                                        # 描述

                                          1. 和归并排序相比，快排是先进行荷兰国旗操作再分开的，而归并是直接二分的。
                                          1. 常数时间来说，快排相比归并和堆排序都要快一些。
                                          1. 各位空间需要0（logN）是递归的需要。

                                        # 需要注意的问题

                                          1. 左边和右边的移动都不只一次所以需要while。并且要注意左边和右边都不能越过对方。
                                          1. 最左边去标准值的时候，最右边先驱动。

                                        因为有可能标准值就在左边，如果左边先驱动因为是<=的，所以有可能不能满足标准值在最右边。

                                        # 稳定性

                                        不稳，因为随时可能会随机交换，随便就会把稳定性排破坏。

                                        # 描述

                                        由于链表只能够从一端开始，所以荷兰国旗的方式最好是这样走。

                                        # 需要注意的问题

                                          1. 是星星移动了，再和三角形指针交换的。所以TEMP永远都在最左边,本质就是荷兰国旗思想，所有小于的放在左边，大于的放在右边。
                                          1. 是要递归的

                                        • .代码：

                                        必须把尾巴也传进去，因为是换数字，所以本质和数组一样，需要有一个位置区分大于区，小于区，和等于区。

                                        list Quick(list begin,list end)
{
	if(begin==NULL||begin==end||end==NULL) return begin; //不应该出现在这里的情况全部递归回去。 
	//获取排序的数值 
	int temp = begin->x;
	
	list F = begin->next;
	list less = begin;
	while(F != end->next && F )   //判断条件也不是空，是不到结尾，根据函数给定的参数 
	{
		if(F->x < temp)
		{
			less = less->next;
			if(less != F)
			{
				int temp = F->x;
				F->x = less->x;
				less->x = temp;
			}
		}
		F = F->next;
	 } 
	
	if(begin!=less)
	{
		int temp = begin->x;
		begin->x = less->x;
		less->x = temp;
	}
	
	Quick(begin,less);
	Quick(less->next,end);
	
	return begin;
}
